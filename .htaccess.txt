# Enable URL rewriting
RewriteEngine On

# Prevent directory listing
Options -Indexes

# Handle 404 errors
ErrorDocument 404 /views/404.php

# Protect sensitive files
<FilesMatch "^\.">
    Order allow,deny
    Deny from all
</FilesMatch>

# Disable access to config directory
<FilesMatch "^config/">
    Order allow,deny
    Deny from all
</FilesMatch>

# PHP error handling
php_flag display_errors off
php_value error_log logs/error.log
